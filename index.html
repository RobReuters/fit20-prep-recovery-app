function startAnalogy(sessions) { const gender = sessionData.genders[0]; // 1. Haal instellingen op (met fallback naar standaardwaarden als de sheet nog niet geladen is) const weightSetting = analogySettings.find(s => s[0] === 'avg_weight_per_session' && s[1] === gender); const distSetting = analogySettings.find(s => s[0] === 'avg_dist_per_session' && s[1] === gender); const weightPerSession = weightSetting ? parseFloat(weightSetting[2]) : (gender === 'v' ? 252.5 : 312.5); const distPerSession = distSetting ? parseFloat(distSetting[2]) : 30; const totalKg = Math.round(sessions * weightPerSession); const totalMeters = Math.round(sessions * distPerSession); // 2. Dieren berekenen (Tier-logica) const set = animalSets.filter(a => a[0] === gender); if (set.length < 3) return console.error("Onvoldoende dieren gevonden in Animal_Sets"); const l_kg = parseInt(set.find(a => a[1] === 'large')[4]); const m_kg = parseInt(set.find(a => a[1] === 'mid')[4]); const s_kg = parseInt(set.find(a => a[1] === 'small')[4]); let rem = totalKg; const l_count = Math.floor(rem / l_kg); rem %= l_kg; const m_count = Math.floor(rem / m_kg); rem %= m_kg; const s_count = Math.floor(rem / s_kg); // 3. Teller animatie KG let curKg = 0; const weightValEl = document.getElementById('weightVal'); const kgInt = setInterval(() => { curKg += totalKg / 40; if(weightValEl) weightValEl.innerText = Math.round(curKg).toLocaleString() + " kg"; if (curKg >= totalKg) { clearInterval(kgInt); if(weightValEl) weightValEl.innerText = totalKg.toLocaleString() + " kg"; } }, 40); // 4. Dieren visueel toevoegen const display = document.getElementById('animalDisplay'); if(display) display.innerHTML = ""; // Clear placeholder const addAnimals = (count, icon, cls) => { for(let i=0; i < Math.min(count, 40); i++) { // Max 40 per soort om browser te sparen setTimeout(() => { let span = document.createElement('span'); span.className = 'animal-icon ' + cls; span.innerText = icon; if(display) display.appendChild(span); }, i * 80); } }; addAnimals(l_count, set.find(a => a[1] === 'large')[5], 'large'); addAnimals(m_count, set.find(a => a[1] === 'mid')[5], 'mid'); addAnimals(s_count, set.find(a => a[1] === 'small')[5], 'small'); // 5. Metafoor zoeken op basis van afstand const meta = distanceMetaphors .filter(m => m[0] === gender || m[0] === 'both') .filter(m => parseInt(m[3]) <= totalMeters) .sort((a,b) => b[3] - a[3])[0] || distanceMetaphors[0]; // 6. Reveal tekst (NL: Kolom 4 & 1 | EN: Kolom 5 & 2) setTimeout(() => { const reveal = document.getElementById('distReveal'); if(reveal && meta) { reveal.innerText = (sessionData.lang === 'en' ? `That's like ${meta[5]} the ${meta[2]}!` : `Dat heb je net zo ${meta[4]} als ${meta[1]}!`); reveal.classList.add('show'); } }, 4000); }
